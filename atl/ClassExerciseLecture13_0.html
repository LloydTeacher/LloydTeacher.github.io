function addMovie(){
	let elements = document.querySelectorAll("input");
	let display = "";
	elements.forEach( item=> {  
		if(item.checked){
			display += item.value +", ";
		}
	});
	if(display.length){
		if(elements[0].value.length){
			display = elements[0].value + " - " + display.replace(/,\s+$/gm,"");
		}else{
			display = display.replace(/,\s+$/gm,"");
		}
	}else{
		display = elements[0].value;
	}

	if(display.length){
		let unorderlist = document.querySelector("ul");
		let li = document.createElement('li');
		li.textContent = display;
		unorderlist.appendChild(li);
	}
}
</script>
