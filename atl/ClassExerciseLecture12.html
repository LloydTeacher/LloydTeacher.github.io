package simplelambo;

import java.util.concurrent.TimeUnit;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;

public class Application {

	public static void main(String[] args) {
		System.setProperty("webdriver.chrome.driver", "C:\\Users\\Lloyd\\Documents\\Automated\\chromedriver.exe");
		// open the browser
		WebDriver driver = new ChromeDriver();
		// navigate to autotrader
		driver.get("https://www.autotrader.ca");
		
		// find the make dropdown
		By SelectBoxLocator = By.id("rfMakes");
		WebElement SelectBox = driver.findElement(SelectBoxLocator);
		
		// select Lamborghini in the make dropdown
		Select dropdown = new Select(SelectBox);
		dropdown.selectByValue("Lamborghini");
		
		// implicit wait
		// waiting for a web page to load
		driver.manage().timeouts().implicitlyWait(4, TimeUnit.MINUTES);
		
		// explicit wait
		// waiting for a element to be available
		
		// find the model dropdown
		By SelectBox2Locator = By.id("rfModel");
		WebElement SelectBox2 = driver.findElement(SelectBox2Locator);
		
		//select huracan in the model dropdown
		Select listbox = new Select(SelectBox2);
		listbox.selectByVisibleText("Huracan");
		
		// find textbox postal code 
		By postalcodelocator = By.cssSelector("#locationAddress");
		WebElement postalcode = driver.findElement(postalcodelocator);
		// write postal code of matrix college
		postalcode.sendKeys("H3H 1E3");
		
		// finding the link sear and click on it
		By searchLocator = By.linkText("Search");
		driver.findElement(searchLocator).click();
		
		// Capture the tile
		String actualtext = driver.getTitle();
		
		// verify that the tile contain New & Used
		if(actualtext.contains("New & Used")) {
			System.out.println("Test is passed");
		}else {
			System.err.println("Test is failed");
		}
		
		driver.quit();
	}

}
