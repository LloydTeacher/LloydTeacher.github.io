package weathershopper;

import org.openqa.selenium.WebDriver;

public class ProductPage {
	
	WebDriver driver;
	
	void verifyTitle() {
		// capture the title
		String actual = driver.getTitle();
		
		boolean verification = actual.contentEquals("Current Temperature");
		
		if (verification) {
			System.out.println("pass");
		}else {
			System.out.println("fail");
		}
		
	}
	
	ProductPage(WebDriver initialDriver){
		driver = initialDriver;
	}

}

package weathershopper;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;

public class HomePage {
	
	WebDriver driver;

	void navigate() {
		driver.get("https://weathershopper.pythonanywhere.com/");
	}
	
	void verifyTitle() {
		// capture the title
		String actual = driver.getTitle();
		
		boolean verification = actual.contentEquals("Current Temperature");
		
		if (verification) {
			System.out.println("pass");
		}else {
			System.out.println("fail");
		}
		
	}
	
	void clickbutton() {
		// find the button
		By buttonLocator = By.linkText("Buy sunscreens");
		WebElement button = driver.findElement(buttonLocator);
		
		// click
		button.click();	
	}
	
	
	HomePage(WebDriver initialDriver){
		driver = initialDriver;
	}
}
