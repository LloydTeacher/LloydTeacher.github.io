package testngmovies;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.annotations.AfterTest;
import org.testng.annotations.BeforeClass;
import org.testng.annotations.BeforeTest;
import org.testng.annotations.Test;

public class Movies {
	
	WebDriver driver;
	WebElement action;
	WebElement serie;
	WebElement bollywood;
	
	@BeforeTest
	void setpath() {
		System.setProperty("webdriver.chrome.driver", "C:\\Users\\Lloyd\\Documents\\Automated\\chromedriver.exe");
	}
	
	@BeforeClass
	void openchrome() {
				// open the browser
		driver = new ChromeDriver();
	}

	@Test(priority = 2)
	void navigate() {
		// navigate to movies
		driver.get("http://127.0.0.1/Automated/movies.html");
	}
	
	@Test(priority = 3)
	void addMovies() {
		driver.findElement(By.xpath("/html/body/form/input[1]")).sendKeys("shaktiman");
		
		action = driver.findElement(By.id("action"));
		action.click();
		
		serie = driver.findElement(By.name("serie"));
		serie.click();
	}
	
	@Test(priority = 4)
	void verification() {

		// verify that radio button action is selected
		if(action.isSelected()) {
			System.out.println("radio button action is selected");
		}else {
			System.err.println("radio button action is not selected");
		}
		
		WebElement drama = driver.findElement(By.id("drama"));

		// verify that radio button drama is selected
		if(drama.isSelected()) {
			System.out.println("radio button drama is selected");
		}else {
			System.err.println("radio button drama is not selected");
		}
		
		// verify that radio button drama is enable
		if(drama.isEnabled()) {
			System.out.println("radio button drama is Enabled");
		}else {
			System.err.println("radio button drama is not Enabled");
		}
		
		// verify that radio button drama is Displayed
		if(drama.isDisplayed()) {
			System.out.println("radio button drama is Displayed");
		}else {
			System.err.println("radio button drama is not Displayed");
		}
		
		// verify that the checkbox serie is selected
		if(serie.isSelected()) {
			System.out.println("checkbox serie is selected");
		}else {
			System.err.println("checkbox serie is not selected");
		}
		
		// verify that the checkbox serie is displayed
		if (serie.isDisplayed()) {
			System.out.println("checkbox serie is Displayed");
		}else {
			System.err.println("checkbox serie is not Displayed");
		}
		
		bollywood = driver.findElement(By.name("bollywood"));

		// verify that the checkbox bollywood is selected
		if(bollywood.isSelected()) {
			System.out.println("checkbox bollywood is selected");
		}else {
			System.err.println("checkbox bollywood is not selected");
		}

	}
	
	
	@AfterTest
	void closeChrome() {
		
		driver.quit();
		
	}
}
