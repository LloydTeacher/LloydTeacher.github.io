// file :  Game
package cricket;

public class Game {
	// properties
	String HomeTeam;
	String AwayTeam;
	int run;
	int wickets;
	double overs;
	
	// method
	void displayResult() {
		System.out.println(
	HomeTeam + " vs " + AwayTeam + " : " + run + " (" + wickets+ " wkts, "+ overs + " Ov)");
		
	}
	
	// constructor
	Game(){
		System.out.println("constructor");
		HomeTeam = "Australia";
		System.out.println(HomeTeam);
		run = 338;
		System.out.println(run);
	}
	
	Game(String initialHomeTeam, String initialAwayTeam, 
			int initialRun, int initialWickets, double initialOvers){
		HomeTeam = initialHomeTeam;
		AwayTeam = initialAwayTeam;
		run = initialRun;
		wickets = initialWickets;
		overs = initialOvers;
	}
}

// file :  Application
package cricket;

public class Application {
	
	public static void main(String[] args) {
		// instantiation
		Game FirstInnings = new Game("Australia", "India", 338,10, 105.4);
		// Australia vs India : 338 (10 wkts, 105.4 Ov)
		// call method
		FirstInnings.displayResult();
		
		Game SecondInnings = new Game("India", "Australia", 244, 10, 100.4);
		// India vs  Australia : 244 (10 wkts, 100.4 Ov)
		// display a variable content
		//System.out.println(MySecondObject.HomeTeam);
		// call a method
		SecondInnings.displayResult();
		
		System.out.println(FirstInnings.run-SecondInnings.run);
		
	}
}
