package firstScript;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class Script {

	public static void main(String[] args) {
		// indicate the path of chrome connector
		System.setProperty("webdriver.chrome.driver", "C:\\Users\\Lloyd\\Documents\\Automated\\chromedriver.exe");
        
		// launch google chrome
		WebDriver driver = new ChromeDriver();
		
		// navigate to https://www.youtube.com
		driver.get("https://www.youtube.com");
		
		// capture the title of the webpage : Find the element title and return the content
		String title = driver.getTitle();

		// write the title in the console
		System.out.println(title.length());

		// close the current tab of google chrome
		driver.close();
	}
}
