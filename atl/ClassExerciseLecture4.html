package triangle;

public class Calculation {
	
	public static void main(String[] args) {
		System.out.println("Triangle");
		int base = 5;
		int height = 10;
		System.out.println("base is "+ base);
		System.out.println("height is "+ height);
		// Exercise
		/*
		 display in the console the surface area of this right angle triangle
		 with base = 5 and height = 10
		 area = base * height /2
		 */
		int surfaceArea = base * height /2;
		String text = "surface area is "+ surfaceArea;
		System.out.println(text);
		
		// Exercise
		/*
		 Display in the console the longest side of a right-angled triangle
		 */
		double hypothenuse = Math.sqrt( Math.pow(base, 2) + Math.pow(height, 2));
        
        // same Calculation but without Math.pow
		hypothenuse = Math.sqrt( base*base + height*height);
		
        // same Calculation but splitting the power
		double hypothenusePowerOftwo = base*base + height*height;
		hypothenuse = Math.sqrt(hypothenusePowerOftwo);
		
		System.out.println("hypothenuse is " + hypothenuse);
	}

}
