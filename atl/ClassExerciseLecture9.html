package alertbox;

import org.openqa.selenium.Alert;
import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class Application {

	public static void main(String[] args) {
		// indicate the path of chrome connector
		System.setProperty("webdriver.chrome.driver", "C:\\Users\\Lloyd\\Documents\\Automated\\chromedriver.exe");
		// launch google chrome
		ChromeDriver driver = new ChromeDriver();
		// Use selenium to navigate to the webpage
		// http://localhost/Automated/alertbox.html
		driver.get("http://localhost/Automated/alertbox.html");
		// find the button
		By buttonLocator = By.tagName("button");
		WebElement button = driver.findElement(buttonLocator);
		
		// click on the button
		button.click();
		
		// Find the alter box
		Alert alertwindow = driver.switchTo().alert();
		
		// Capture the text inside alert box
		String actualtext = alertwindow.getText();
		
		// Display the text in the console
		System.out.println(actualtext);
		
		// click on ok button
		alertwindow.accept();
		
		// Find the button show yes no
		By yesNoLocator = By.id("okcancel");
		WebElement yesNoButton = driver.findElement(yesNoLocator);
		
		// click on the button show yes no
		yesNoButton.click();
		
		// Find the alter box
		alertwindow = driver.switchTo().alert();
		
		// click on cancel button
		alertwindow.dismiss();
		
	}
}
